Installation

inject the module and prepare your routes

1 Include Angumate

...
  <script src="//ajax.googleapis.com/ajax/libs/angularjs/1.0.4/angular.min.js"></script>
  <script src="app.js"></script>
  <script src="angumate.min.js"></script>
</body>
</html>

2 Inject Angumate to your main module

var app = angular.module('app', [Angumate]);

3 Add Angumate as a promise to your routes

.when('/home', {
  templateUrl: 'partials/home.html',
  resolve: ['angumate', function(angumate) {
    return angumate.promise();
  }]
})

or

.when('/download', {
  templateUrl: 'partials/download.html',
  resolve: {
    angumate: ['angumate', function(angumate) {
      return angumate.promise();
    }]
  }
})
